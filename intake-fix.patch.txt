+export type IntakePayload = {
+  form_type: string;
+  lane: 'strategic_transformations' | 'fractional_lane' | 'uri' | 'internal';
+  offer_type: string;
+  vetted: string;
+  first_name: string;
+  last_name: string;
+  email: string;
+  phone: string;
+  company: string;
+  title: string;
+  audience_type: string;
+  investment_readiness: string;
+  timeline: string;
+  primary_need: string;
+  summary_notes: string;
+  source_url: string;
+  submitted_at: string;
+  [key: string]: string | string[];
+};
+
+const PUBLIC_N8N_WEBHOOK_URL = (import.meta.env.PUBLIC_N8N_WEBHOOK_URL || '').trim();
+
+export function splitFullName(fullName: string): { firstName: string; lastName: string } {
+  const cleanName = String(fullName || '').trim().replace(/\s+/g, ' ');
+  if (!cleanName) return { firstName: '', lastName: '' };
+
+  const parts = cleanName.split(' ');
+  const firstName = parts.shift() || '';
+  const lastName = parts.join(' ');
+
+  return { firstName, lastName };
+}
+
+export async function sendIntake(payload: IntakePayload) {
+  if (!PUBLIC_N8N_WEBHOOK_URL) {
+    throw new Error('Missing PUBLIC_N8N_WEBHOOK_URL');
+  }
+
+  const response = await fetch(PUBLIC_N8N_WEBHOOK_URL, {
+    method: 'POST',
+    headers: { 'Content-Type': 'application/json' },
+    body: JSON.stringify(payload),
+    mode: 'cors',
+  });
+
+  if (!response.ok) {
+    throw new Error(`Webhook rejected payload with status ${response.status}`);
+  }
+}
